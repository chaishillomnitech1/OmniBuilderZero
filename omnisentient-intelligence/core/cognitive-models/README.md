## Conflict Resolution

### Conflict Types

| Type | Description | Resolution Strategy |
|------|-------------|---------------------|
| **Resource Conflict** | Multiple goals need same resources | Priority-based allocation |
| **Time Conflict** | Overlapping deadlines | Sequence optimization |
| **Value Conflict** | Goals support different values | Value hierarchy resolution |
| **Dependency Conflict** | Circular or blocking dependencies | Dependency analysis and restructuring |

## Goal API

### Create Goal

```javascript
// Create a new goal
async function createGoal(goalSpec) {
  // Validate specification
  validateGoalSpec(goalSpec);
  
  // Generate ID
  const goal = {
    id: generateGoalId(),
    ...goalSpec,
    state: 'pending',
    progress: {
      percentage: 0,
      milestones_completed: 0,
      milestones_total: goalSpec.sub_goals?.length || 1,
      last_update: new Date().toISOString()
    },
    metadata: {
      created_at: new Date().toISOString(),
      created_by: getCurrentAgentId()
    }
  };
  
  // Calculate initial priority
  goal.priority.combined_score = calculatePriority(goal);
  
  // Store goal
  await goalStore.save(goal);
  
  // Create sub-goals if specified
  if (goalSpec.sub_goals) {
    for (const subGoalSpec of goalSpec.sub_goals) {
      await createGoal({
        ...subGoalSpec,
        parent_goal: goal.id
      });
    }
  }
  
  return goal;
}
```

```javascript
// Update goal progress
async function updateProgress(goalId, progressUpdate) {
  const goal = await goalStore.get(goalId);
  
  // Update progress
  goal.progress = {
    ...goal.progress,
    ...progressUpdate,
    last_update: new Date().toISOString()
  };
  
  // Check for state transitions
  if (goal.progress.percentage >= 100) {
    if (meetsSuccessCriteria(goal)) {
      goal.state = 'completed';
      await handleGoalCompletion(goal);
    }
  }
  
  // Update parent goal if exists
  if (goal.parent_goal) {
    await updateParentProgress(goal.parent_goal);
  }
  
  // Save and notify
  await goalStore.save(goal);
  await notifyProgressUpdate(goal);
  
  return goal;
}
```

### Select Next Goal

```javascript
// Select next goal to work on
async function selectNextGoal(agentId) {
  const agent = await agentStore.get(agentId);
  const availableGoals = await getAssignedGoals(agentId, {
    states: ['pending', 'active'],
    not_blocked: true
  });
  
  // Calculate effective priority for each goal
  const scoredGoals = availableGoals.map(goal => ({
    goal,
    score: calculateEffectivePriority(goal, agent)
  }));
  
  // Sort by score
  scoredGoals.sort((a, b) => b.score - a.score);
  
  // Apply motivation check
  const topGoal = scoredGoals[0];
  const motivation = calculateMotivation(agent, topGoal.goal);
  
  if (motivation < config.minimum_motivation) {
    // Find alternative with better motivation
    for (const scored of scoredGoals.slice(1)) {
      const altMotivation = calculateMotivation(agent, scored.goal);
      if (altMotivation >= config.minimum_motivation) {
        return scored.goal;
      }
    }
  }
  
  return topGoal.goal;
}
```
